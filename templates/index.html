🔵 BFS — BREADTH FIRST SEARCH
# BFS — O(V+E) — Queue-based
def bfs_trace(graph, source):
  visited = {}
  queue = [(source, 0)]
  chain = []

  while queue:
    node, depth = queue.pop(0) # FIFO
    if node in visited: continue
    if depth > incubation: continue
    visited[node] = depth
    chain.append(node)
    for nbr in graph[node]:
      queue.append((nbr, depth+1))
  return chain

# Time: O(V+E) · Space: O(V)
# Finds shortest path to all contacts
🔴 DFS — DEPTH FIRST SEARCH
# DFS — O(V+E) — Stack-based
def dfs_trace(graph, source):
  visited = {}
  stack = [(source, 0)]
  chain = []

  while stack:
    node, depth = stack.pop() # LIFO
    if node in visited: continue
    if depth > incubation: continue
    visited[node] = depth
    chain.append(node)
    for nbr in reversed(graph[node]):
      stack.append((nbr, depth+1))
  return chain

# Time: O(V+E) · Space: O(V)
# Explores deep branches first
